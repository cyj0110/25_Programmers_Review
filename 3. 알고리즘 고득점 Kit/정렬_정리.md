def solution(array, commands):
    answer = []
    
    for command in commands:
        temp = []
        i, j, k = command[0], command[1], command[2]
        temp = sorted(array[i-1:j])
        answer.append(temp[k-1])
              
    return answer

if __name__ == "__main__":
    array = input()
    commands = input()
    
    result = solution(array, commands)
    print(result)
